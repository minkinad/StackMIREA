```python title="Notebook5.ipynb (класс Rational)"
class Rational:
    @staticmethod
    def gcd(a, b):
        while b != 0:
            a, b = b, a % b
        return a

    def __init__(self, n, d):
        z = 1 if n * d >= 0 else -1
        n, d = abs(n), abs(d)
        k = self.gcd(n, d)
        self.num = z * n // k
        self.den = d // k
```
